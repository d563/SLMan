vim
===
::

    yum -y install vim-enhanced

.. index:: /etc/profile.d/vim.sh

::

    mv /etc/profile.d/vim.sh /etc/profile.d/vim.sh.org
    echo "alias vi='vim'
    alias view='vim -R'" > /etc/profile.d/vim.sh

.. index:: /etc/vimrc

::

    cat << '[EOF]' >> /etc/vimrc
    set expandtab
    set tabstop=4
    set softtabstop=4
    set shiftwidth=4
    " set autoindent
    set ignorecase
    set incsearch
    set ruler
    [EOF]
