mysql
=====
::

    yum -y install mysql-server

.. index:: /etc/my.cnf

::

    sed -i.org \
    -e '/\[mysqld\]/a\
    character-set-server=utf8' \
    /etc/my.cnf

    echo '
    [client]
    default-character-set=utf8' >> /etc/my.cnf

    /etc/init.d/mysqld start

::

    mysql_secure_installation
    mysql -u root -p
