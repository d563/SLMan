::

    mkdir ~/.ssh
    cd ~/.ssh
    ssh-keygen -t rsa -C "mail.d563@gmail.com"
    chmod 600 id_rsa
    chmod 644 id_rsa.pub
    ssh -T git@github.com
    # cat id_rsa.pub >> authorized_keys
