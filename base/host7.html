1.2.7. firewalld¶

cat /usr/lib/firewalld/services/ssh.xml
cat /usr/lib/firewalld/services/https.xml
firewall-cmd --list-all
firewall-cmd --permanent --add-service=https
firewall-cmd --permanent --remove-service=ssh
firewall-cmd --reload
