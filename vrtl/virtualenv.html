8.1. virtualenv¶

pip install virtualenv virtualenvwrapper

mkdir /usr/local/lib/virtualenv
echo "WORKON_HOME=/usr/local/lib/virtualenv
. $(which virtualenvwrapper.sh)" > /etc/profile.d/virtualenv.sh

. /etc/profile.d/virtualenv.sh

8.1.1. usage¶

mkvirtualenv d563
workon d563
deactivate
rmvirtualenv d563
