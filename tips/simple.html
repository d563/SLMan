9.4.2. find¶

find ./ \
  -mindepth 1 \
  -maxdepth 2 \
  -type f \
  -name "*.zip" \
  -user d563 \
  -group d563 \
  -ctime +90 \
  -perm 664 \
  -exec echo {} \; | xargs echo
