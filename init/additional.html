1.2.4. vim¶

mv /etc/profile.d/vim.sh /etc/profile.d/vim.sh.org
echo "alias vi='vim'
alias view='vim -R'" > /etc/profile.d/vim.sh

cat << '[EOF]' >> /etc/vimrc
set expandtab
set tabstop=4
set softtabstop=4
set shiftwidth=4
" set autoindent
set ignorecase
set incsearch
set ruler
[EOF]
