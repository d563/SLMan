#!/bin/awk -f
# template.awk
BEGIN {
  FS = ",";
  OFS = "/";
  i = 0;
}
/str/ {
  i++;
}
{
  if ($0 != "") {
    printf "line:%d -> %s\n", NR, $0;
  }
}
END {
  print i, NR;
}

cat << '[EOF]' >> dat
a,b,c
d,str,f

g,h,i
[EOF]

chmod +x ./template.awk
./template.awk dat
