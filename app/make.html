#ifndef HEADER_H
#define HEADER_H
/* header.h */
typedef struct SLMan {
  void (*exec)(struct SLMan *sl);
  char name[8];
} SLMan;
void sub(SLMan *sl);
#endif

#include <stdio.h>
#include "header.h"
/* sub.c */
void sub(SLMan *sl) {
  printf("%s\n", sl->name);
}

#include <stdio.h>
#include "header.h"
/* main.c */
int main(int argc, char *argv[]) {
  SLMan sl;
  printf("%d\n", sizeof(sl));
  sl.exec = sub;
  fgets(sl.name, sizeof(sl.name), stdin);
  sl.exec(&sl);
  return 0;
}

# makefile
DEST = main
OBJS = main.o sub.o
${DEST}: header.h ${OBJS}
    gcc -Wall -O2 -o ${DEST} ${OBJS}
.c.o:
    gcc -c $<
clean:
    rm -f ${DEST} ${OBJS}
